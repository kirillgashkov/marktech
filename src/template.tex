  \appendices%
  \renewcommand{\appendixname}{\template@header@appendix@internal@appendixname}%
  \renewcommand{\thesection}{\Asbuk{section}}%
  %
  \titleformat{\section}[display]{\filcenter\bfseries\Large}{\appendixname~\thesection}{2ex}{\vspace{-0.5\baselineskip}}%
  \titlespacing{\section}{0pt}{3ex}{2ex}%
  \titleformat{\subsection}[block]{\template@header@default@internal@align\bfseries\large}{\thesubsection}{0.5em}{}%
  \titlespacing{\subsection}{\template@header@appendix@internal@margin@left}{3ex}{2ex}%

% Table

\usepackage{longtable}


% Variable width \hline and \cline. See https://tex.stackexchange.com/a/24552.
% booktabs was also considered, their \specialrule worked great but \cmidrule
% not so much, it had problems with vertical lines.

\makeatletter
\newlength{\template@table@internal@arrayrulewidth}
\newcommand{\template@table@arrayrulewidth@set}[1]{%
  \noalign{\global\setlength{\arrayrulewidth}{#1}}%
}
\newcommand{\template@table@arrayrulewidth@save}{%
  \noalign{\global\setlength{\template@table@internal@arrayrulewidth}{\arrayrulewidth}}%
}
\newcommand{\template@table@arrayrulewidth@restore}{%
  \noalign{\global\setlength{\arrayrulewidth}{\template@table@internal@arrayrulewidth}}%
}
\makeatother

\makeatletter
\newcommand{\varhline}[1]{
  \template@table@arrayrulewidth@save%
  \template@table@arrayrulewidth@set{#1}%
  \hline%
  \template@table@arrayrulewidth@restore%
}
\newcommand{\varcline}[2]{%
  \template@table@arrayrulewidth@save%
  \template@table@arrayrulewidth@set{#1}%
  \cline{#2}%
  \template@table@arrayrulewidth@restore%
}
\makeatother
